#[derive(Serialize, Deserialize, Clone, Debug)]
pub struct AisStreamMessage {
    #[serde(rename = "Message")]
    pub message: BTreeMap<String, Value>,
    #[serde(rename = "MessageType")]
    pub message_type: AisMessageType,
    #[serde(rename = "MetaData")]
    pub metadata: AisMetadata,
}

impl AisStreamMessage {
    pub fn body_as<T>(&self) -> Result<Option<T>, serde_json::Error>
    where
        T: DeserializeOwned,
    {
        self.message
            .get(self.message_type.as_str())
            .cloned()
            .map(serde_json::from_value)
            .transpose()
    }
}

#[derive(Clone, Debug, PartialEq, Eq)]
pub enum AisMessageType {
    UnknownMessage,
    ShipStaticData,
    StandardClassBPositionReport,
    PositionReport,
    Other(String),
}

impl AisMessageType {
    pub fn as_str(&self) -> &str {
        match self {
            Self::UnknownMessage => "UnknownMessage",
            Self::ShipStaticData => "ShipStaticData",
            Self::StandardClassBPositionReport => "StandardClassBPositionReport",
            Self::PositionReport => "PositionReport",
            Self::Other(value) => value.as_str(),
        }
    }
}

impl Serialize for AisMessageType {
    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
    where
        S: Serializer,
    {
        serializer.serialize_str(self.as_str())
    }
}

impl<'de> Deserialize<'de> for AisMessageType {
    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
    where
        D: Deserializer<'de>,
    {
        let value = String::deserialize(deserializer)?;

        Ok(match value.as_str() {
            "UnknownMessage" => Self::UnknownMessage,
            "ShipStaticData" => Self::ShipStaticData,
            "StandardClassBPositionReport" => Self::StandardClassBPositionReport,
            "PositionReport" => Self::PositionReport,
            _ => Self::Other(value),
        })
    }
}

pub(crate) fn run_ais(_conn: DbConnection) -> Result<(), Box<dyn Error>> {
    // Aisstream
    let aisstream_api_url: Uri = Uri::from_static("wss://stream.aisstream.io/v0/stream");
    let aisstream_api_key: String = std::env::var("AISSTREAM_API_KEY").unwrap_or("401620aea8c9f66129af3d8b1caec95a86144a61".to_owned());

    // WebSocket
    println!("Connecting `{aisstream_api_url}`");
    let (mut socket, _) = tungstenite::connect(aisstream_api_url)?;

    let auth_request = AuthRequest {
        api_key: aisstream_api_key.clone(),
        bounding_boxes: vec![[[51.809685,3.931732], [52.041860, 4.619751]]],
    };

        // bounding_boxes: vec![[[3.931732, 51.809685], [4.619751, 52.041860]]],


    let auth_request_bytes = Bytes::copy_from_slice(&serde_json::to_vec(&auth_request)?);

    // let message = serde_json::json!({
    //         "APIKey": aisstream_api_key,
    //         "BoundingBoxes": [[[-180, -90], [180, 90]]]
    //     });

    // Authentication
    println!("Authenticating...");
    socket.send(tungstenite::Message::Binary(
        auth_request_bytes
    ))?;
    let message = match socket.read()? {
        tungstenite::Message::Binary(message) => {
            println!("Received authentication response: {:?}", message);
            match serde_json::from_slice::<AuthMessage>(&message)? {
                AuthMessage::AuthError(message) => {
                    return Err(format!("Authentication error: {message:?}").into());
                }
                AuthMessage::Message(message) => Some(message),
            }
        }
        _ => None,
    };
    println!("Successfully authenticated");
    if let Some(message) = message {
        print_message(&message);
    }

    // Loop
    loop {
        println!("Waiting for messages...");
        let message = match socket.read()? {
            tungstenite::Message::Binary(message) => {
                serde_json::from_slice::<AisStreamMessage>(&message)?
            }
            tungstenite::Message::Close(message) => {
                return Err(format!("Connection closed: {message:?}").into());
            }
            _ => continue,
        };
        print_message(&message);
    }
}

fn print_message(message: &AisStreamMessage) {
    match &message.message_type {
        AisMessageType::PositionReport => {
            if let Ok(Some(report)) = message.body_as::<PositionReport>() {
                            add_location_report(
                &DbConnection::current().expect("No database connection available"),
                message.metadata.mmsi,
                message.metadata.latitude,
                message.metadata.longitude,
                Some(report.cog),
                Some(report.sog),
            ).unwrap_or_else(|e| eprintln!("Error adding location report: {e}"));

                println!(
                    "{} position: lat={}, lon={}, sog={}, cog={}",
                    message.metadata.ship_name,
                    report.latitude,
                    report.longitude,
                    report.sog,
                    report.cog
                );
            }
        }
        AisMessageType::StandardClassBPositionReport => {
            if let Ok(Some(report)) = message.body_as::<StandardClassBPositionReport>() {
                add_location_report(
                    &DbConnection::current().expect("No database connection available"),
                    message.metadata.mmsi,
                    message.metadata.latitude,
                    message.metadata.longitude,
                    Some(report.cog),
                    Some(report.sog),
                ).unwrap_or_else(|e| eprintln!("Error adding location report: {e}"));

                println!(
                    "{} class B position: lat={}, lon={}, cog={}, sog={}",
                    message.metadata.ship_name,
                    report.latitude,
                    report.longitude,
                    report.cog,
                    report.sog
                );
            }
        }
        AisMessageType::ShipStaticData => {
            if let Ok(Some(report)) = message.body_as::<ShipStaticData>() {
                add_ship(
                    &DbConnection::current().expect("No database connection available"),
                    report.name.clone(),
                    Some(report.call_sign.clone()),
                ).unwrap_or_else(|e| eprintln!("Error adding ship: {e}"));
                println!(
                    "{} static data: callsign={}, destination={}",
                    report.name, report.call_sign, report.destination
                );
            }
        }
        AisMessageType::UnknownMessage => {
            let _ = message.body_as::<UnknownMessage>();
            println!("{} unknown AIS message", message.metadata.ship_name);
        }
        AisMessageType::Other(_) => {}
    }

    let pretty_message = serde_json::to_string_pretty(message).unwrap_or_else(|_| "Failed to serialize message".to_string());
    println!("{pretty_message}");
}

#[derive(Serialize, Deserialize, Clone, Debug)]
pub struct AuthRequest {
    #[serde(rename = "APIKey")]
    pub api_key: String,
    #[serde(rename = "BoundingBoxes")]
    pub bounding_boxes: Vec<[[f64; 2]; 2]>,
}

#[derive(Serialize, Deserialize, Clone, Debug)]
#[serde(untagged)]
pub enum AuthMessage {
    AuthError(AuthError),
    Message(AisStreamMessage),
}

#[derive(Serialize, Deserialize, Clone, Debug)]
pub struct AuthError {
    pub error: String,
}

impl fmt::Display for AuthError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        self.error.fmt(f)
    }
}
